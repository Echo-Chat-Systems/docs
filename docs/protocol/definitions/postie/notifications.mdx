<ParamsList>
    <Param name="mode" type="string" required>
        Notifications mode. Can be one of the following: "all", "mentions", "direct-mentions"

        Controls what type of events raised will trigger a notification to the user. This could be an in-app ping or
        a push notification.
    </Param>
    <Param name="suppressed-roles" type="array" required>
        Role IDs from which notifications are suppressed.
    </Param>
</ParamsList>
