<ParamsList>
    <Param name="banner" type="string">
        Either a URL linking to a valid image file or a [snowflake](/docs/protocol/snowflake) referencing a [media](/docs/protocol/definitions/media/) item.
    </Param>
    <Param name="icon" type="string" required>
        Either a URL linking to a valid image file or a [snowflake](/docs/protocol/snowflake) referencing a [media](/docs/protocol/definitions/media/) item.
    </Param>
    <Param name="css" type="string">
        Additional guild-wide css.
    </Param>
</ParamsList>
